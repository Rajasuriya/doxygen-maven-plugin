103 	protected File getOutputDirectory() {
104 		if (!outputDirectory.isAbsolute()) {
105 			outputDirectory = new File(getBasedir(), outputDirectory.getPath());
106 		}
107 		//Create the folder structure.
108 		if (!outputDirectory.exists()) {
109 			outputDirectory.mkdirs();
110 		}
111 
112 		return outputDirectory.getAbsoluteFile();
113 	}

115     /**
116      * This code is lent from 
117      * http://fisheye.codehaus.org/browse/~raw,r=8882/mojo/trunk/mojo/exec-maven-plugin/src/main/java/org/codehaus/mojo/exec/ExecMojo.java
118      * @return Toolchain instance.
119      */
120     private Toolchain getToolchain()
121     {
122         Toolchain tc = null;
123     
124         try
125         {
126             if ( session != null ) // session is null in tests..
127             {
128                 ToolchainManager toolchainManager =
129                     (ToolchainManager) session.getContainer().lookup( ToolchainManager.ROLE );
130     
131                 if ( toolchainManager != null )
132                 {                    
133                     tc = toolchainManager.getToolchainFromBuildContext( "jdk", session );
134                 }
135             }
136         }
137         catch ( ComponentLookupException componentLookupException )
138         {
139             // just ignore, could happen in pre-2.0.9 builds..
140         }
141         return tc;
142     }

144     /**
145      * This will check if the given information for the doxygen executable 
146      * is enough. If not than we search on the path for doxygen executable.
147      * @return Path to the doxygen executable.
148      */
149     private String getExecutablePath()
150     { 
151         File execFile = new File( executable );
152         if ( execFile.exists() )
153         {
154             getLog().debug( "Toolchains are ignored, 'executable' parameter is set to " + executable );
155             return execFile.getAbsolutePath();
156         } 
157         else
158         {
159             Toolchain tc = getToolchain();
160             
161             // if the file doesn't exist & toolchain is null, the exec is probably in the PATH...
162             // we should probably also test for isFile and canExecute, but the second one is only
163             // available in SDK 6.
164             if ( tc != null ) 
165             {
166                 getLog().info( "Toolchain in doxygen plugin: " + tc );    
167                 executable = tc.findTool( executable );                                                
168             }
169         }
170 
171         return executable;        
172     }

189         }
190 
191         return ret;
192     }

421     /**
422      * This is called for boolean configuration items.
423      * @param config The configuration where to write to.
424      * @param key The configuration parameter name with it's default values etc.
425      * @param value The reals value, base on the Plugin configuration.
426      */
427     private void addConfiguration(PrintWriter config, DoxygenParameters key, boolean value) {
428     	if (value) {
429             addConfiguration(config, key, "YES");
430     	} else {
431     		addConfiguration(config, key, "NO");
432     	}
433     }
434     /**
435      * This is called for integer configuration items.
436      * @param config The configuration where to write to.
437      * @param key The configuration parameter name with it's default values etc.
438      * @param value The reals value, base on the Plugin configuration.
439      */
440     private void addConfiguration(PrintWriter config, DoxygenParameters key, Integer value) {
441     	if (value == null) {
442     		value = Integer.parseInt(key.getDefaultValue());
443     	}
444     	addConfiguration(config, key, value.toString());
445     }
446 
447     /**
448      * This is called for String configuration items.
449      * @param config The configuration where to write to.
450      * @param key The configuration parameter name with it's default values etc.
451      * @param value The reals value, base on the Plugin configuration.
452      */
453     private void addConfiguration(PrintWriter config, DoxygenParameters key, String value) {
454     	if (value == null) {
455     		value = key.getDefaultValue();
456     	} else if ((value.length() == 0) || (value.trim().length() == 0)) {
457     		value = key.getDefaultValue();
458     	}
459 
460     	//If we have quoted parameters.
461     	if (key.getType().equals(DoxygenParameterType.STRING_QUOTED)) {
462 			value = '"' + value + '"';
463     	}
464 
465     	addConfiguration(config, key.getDescription(), key.name(), value);
466     }
